# md-reader 多标签工作区 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 将 `md-reader` 从单文档工作区重构为多标签工作区，落地顶部 Chrome 风格标签栏、每标签独立文档槽、会话级保存闸门与标签恢复链路。

**Architecture:** 先在 `App.tsx` 内部完成 `WorkspaceSession + WorkspaceTab` 状态迁移，并通过兼容映射继续驱动现有 `TopBar/WorkspaceLayout`，保证中间态可编译。再分别替换顶部栏和正文工作区消费面，最后补会话级保存闸门、回包归属和顶栏单行压缩回归。

**Tech Stack:** React 19, TypeScript, Vitest, Testing Library, local bridge access, plain CSS in `src/index.css`

## 关键约束

- 同一路径文档在同一会话中只允许存在一个标签实例。
- `mode` 与 `regularViewState` 是会话级状态，不跟随单个标签保存独立值。
- `persistedContent/draftContent/saveState/saveErrorMessage/mtimeMs/lastKnownScrollTop/headings` 是标签级状态。
- 顶部标签栏必须始终单行，窄宽度下压缩标签宽度，不允许掉到第二行。
- 右侧操作区全部收敛为图标按钮，并保持固定宽度占位。
- 顶部不再出现 `topbar__status`、项目名、`状态：...`、`保存：...` 常驻文案。

import type { RegularViewState, WorkspaceMode } from '../app/TopBar'

export type TabSaveState = 'clean' | 'typing' | 'save_queued' | 'saving_background' | 'save_failed_retryable' | 'conflict_hard' | 'leaving_with_pending_flush'

export interface WorkspaceTab {
  id: string
  documentPath: string
  persistedContent: string | null
  draftContent: string | null
  mtimeMs: number | null
  saveState: TabSaveState
  saveErrorMessage: string | null
  lastKnownScrollTop: number
}

export interface WorkspaceSession {
  tabs: WorkspaceTab[]
  activeTabId: string | null
  mode: WorkspaceMode
  regularViewState: RegularViewState
}

export function getActiveTab(session: WorkspaceSession): WorkspaceTab | null {
  return session.tabs.find((tab) => tab.id === session.activeTabId) ?? null
}

```ts
function openOrActivateTab(documentPath: string) {
  setSession((previous) => {
    const existing = previous.tabs.find((tab) => tab.documentPath === documentPath)
    if (existing) {
      return { ...previous, activeTabId: existing.id }
    }

    const nextTab: WorkspaceTab = {
      id: documentPath,
      documentPath,
      persistedContent: null,
      draftContent: null,
      mtimeMs: null,
      saveState: 'clean',
      saveErrorMessage: null,
      lastKnownScrollTop: 0,
    }

    return {
      ...previous,
      tabs: [...previous.tabs, nextTab],
      activeTabId: nextTab.id,
    }
  })
}
```

```ts
function updateTab(tabId: string, updater: (tab: WorkspaceTab) => WorkspaceTab) {
  setSession((previous) => ({
    ...previous,
    tabs: previous.tabs.map((tab) => (tab.id === tabId ? updater(tab) : tab)),
  }))
}
```

```ts
async function saveTab(tabId: string, reason: 'background' | 'leave'): Promise<boolean> {
  const snapshot = sessionRef.current
  const tab = snapshot.tabs.find((entry) => entry.id === tabId)
  if (!tab || tab.draftContent == null || tab.draftContent === tab.persistedContent) {
    return true
  }

  const saved = await saveDocumentContentToBridge(
    activeProjectIdRef.current!,
    activeProfileIdRef.current,
    tab.documentPath,
    tab.draftContent,
    tab.mtimeMs,
    tab.persistedContent ? createContentHash(tab.persistedContent) : null,
  )

  updateTab(tabId, (current) => ({
    ...current,
    persistedContent: saved.content,
    draftContent: saved.content,
    mtimeMs: saved.mtimeMs,
    saveState: 'clean',
    saveErrorMessage: null,
  }))

  return true
}
```

```ts
type PendingWorkspaceAction =
  | { kind: 'restart-service' }
  | { kind: 'stop-service' }
  | { kind: 'switch-project'; projectId: string }
  | { kind: 'switch-profile'; profileId: string }

const dirtyTabs = session.tabs.filter((tab) => tab.draftContent !== tab.persistedContent)

if (dirtyTabs.length > 0) {
  setPendingWorkspaceAction({ kind: 'restart-service' })
  setShowSessionSaveDialog(true)
  return
}
```
