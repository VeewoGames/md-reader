# 顶部标签页拖拽交互 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 为 `md-reader` 顶部标签栏实现“仅标签栏内部左右重排”的 Chrome-like 拖拽交互，并保证顺序会写回 `session.tabs` 与 `openDocumentPaths` 恢复链路。

**Architecture:** 本轮不引入新窗口、跨容器拖放或多行标签。实现分成三层：`src/workspace/tab-order.ts` 提供纯排序/命中辅助；`src/app/TopBar.tsx` 负责 pointer drag 生命周期、测量、焦点归位和预览位移；`src/App.tsx` 负责提交真实标签顺序并写回本地状态。测试分成纯函数测试、TopBar 组件拖拽测试和 App 恢复链路测试三组。

**Tech Stack:** React 19, TypeScript, Vitest, Testing Library, plain CSS in `src/index.css`

```ts
export interface TabRect {
  left: number
  width: number
  center: number
}

export function moveTabId(order: string[], dragTabId: string, nextIndex: number): string[] {
  const currentIndex = order.indexOf(dragTabId)
  if (currentIndex < 0 || currentIndex === nextIndex) {
    return order
  }

  const next = order.slice()
  next.splice(currentIndex, 1)
  next.splice(nextIndex, 0, dragTabId)
  return next
}

export function findDropIndexByCenter(
  order: string[],
  dragTabId: string,
  dragCenterX: number,
  rectsById: Record<string, TabRect>,
): number {
  const withoutDrag = order.filter((id) => id !== dragTabId)
  const targetIndex = withoutDrag.findIndex((id) => dragCenterX < rectsById[id].center)
  return targetIndex < 0 ? withoutDrag.length : targetIndex
}

export function buildShiftMap({
  committedOrder,
  previewOrder,
  dragTabId,
  rectsById,
}: {
  committedOrder: string[]
  previewOrder: string[]
  dragTabId: string
  rectsById: Record<string, TabRect>
}) {
  return Object.fromEntries(
    committedOrder
      .filter((id) => id !== dragTabId)
      .map((id) => {
        const committedIndex = committedOrder.indexOf(id)
        const previewIndex = previewOrder.indexOf(id)
        const delta = previewIndex - committedIndex
        if (delta === 0) {
          return [id, 0]
        }

        const currentRect = rectsById[id]
        const dragRect = rectsById[dragTabId]
        const slotSize = dragRect.width + (currentRect.left - dragRect.left - dragRect.width)
        return [id, delta * slotSize]
      }),
  )
}
```

```tsx
interface TopBarProps {
  // ...
  onTabReorder?: (nextOrderedTabIds: string[]) => void
}

const DRAG_THRESHOLD_PX = 6
const TAB_GAP_PX = 6

const [dragState, setDragState] = useState<'idle' | 'press_pending' | 'dragging' | 'settling'>('idle')
const [dragTabId, setDragTabId] = useState<string | null>(null)
const [previewOrder, setPreviewOrder] = useState<string[] | null>(null)
const [dragOffsetX, setDragOffsetX] = useState(0)
const settlingTimerRef = useRef<number | null>(null)
const dragMetaRef = useRef<{
  pointerId: number
  startX: number
  order: string[]
  rectsById: Record<string, TabRect>
  triggerElement: HTMLButtonElement | null
} | null>(null)

function handleTabPointerDown(event: React.PointerEvent<HTMLButtonElement>, tabId: string) {
  if (event.button !== 0) return
  dragMetaRef.current = {
    pointerId: event.pointerId,
    startX: event.clientX,
    order: tabs.map((tab) => tab.id),
    rectsById: collectTabRects(),
    triggerElement: event.currentTarget,
  }
  setDragState('press_pending')
  setDragTabId(tabId)
  setPreviewOrder(tabs.map((tab) => tab.id))
}

function beginDragging(meta: NonNullable<typeof dragMetaRef.current>) {
  meta.triggerElement?.setPointerCapture(meta.pointerId)
  window.addEventListener('pointermove', handleWindowPointerMove)
  window.addEventListener('pointerup', handleWindowPointerUp)
  window.addEventListener('pointercancel', handleWindowPointerCancel)
}

function cleanupDragging(pointerId?: number) {
  window.removeEventListener('pointermove', handleWindowPointerMove)
  window.removeEventListener('pointerup', handleWindowPointerUp)
  window.removeEventListener('pointercancel', handleWindowPointerCancel)
  dragMetaRef.current?.triggerElement?.releasePointerCapture(pointerId ?? dragMetaRef.current?.pointerId ?? -1)
}

function handleWindowPointerMove(event: PointerEvent) {
  const meta = dragMetaRef.current
  if (!meta || event.pointerId !== meta.pointerId || dragTabId == null) return

  const offsetX = event.clientX - meta.startX
  if (dragState === 'press_pending') {
    if (Math.abs(offsetX) < DRAG_THRESHOLD_PX) return
    beginDragging(meta)
  }

  const baseOrder = previewOrder ?? meta.order
  const dragCenterX = meta.rectsById[dragTabId].center + offsetX
  const nextIndex = findDropIndexByCenter(baseOrder, dragTabId, dragCenterX, meta.rectsById)
  const nextOrder = moveTabId(baseOrder, dragTabId, nextIndex)

  setDragState('dragging')
  setDragOffsetX(offsetX)
  setPreviewOrder(nextOrder)
}

function finishDrag() {
  const nextOrder = previewOrder
  if (dragState === 'dragging' && dragTabId && nextOrder) {
    onTabReorder?.(nextOrder)
  }
  const pointerId = dragMetaRef.current?.pointerId
  const triggerElement = dragMetaRef.current?.triggerElement ?? null
  cleanupDragging(pointerId)
  dragMetaRef.current = null
  setDragState('settling')
  setDragOffsetX(0)
  setDragTabId(null)
  setPreviewOrder(null)
  triggerElement?.focus()
  if (settlingTimerRef.current != null) {
    window.clearTimeout(settlingTimerRef.current)
  }
  settlingTimerRef.current = window.setTimeout(() => {
    setDragState('idle')
    settlingTimerRef.current = null
  }, 160)
}

function handleWindowPointerUp(event: PointerEvent) {
  if (event.pointerId === dragMetaRef.current?.pointerId) {
    finishDrag()
  }
}

function handleWindowPointerCancel(event: PointerEvent) {
  if (event.pointerId === dragMetaRef.current?.pointerId) {
    finishDrag()
  }
}
```

function reorderSessionTabs(current: WorkspaceSession, nextOrderedTabIds: string[]): WorkspaceSession {
  const tabMap = new Map(current.tabs.map((tab) => [tab.id, tab]))
  const reorderedTabs = nextOrderedTabIds
    .map((id) => tabMap.get(id))
    .filter((tab): tab is WorkspaceTab => tab != null)

  return reorderedTabs.length === current.tabs.length
    ? { ...current, tabs: reorderedTabs }
    : current
}

async function handleTabReorder(nextOrderedTabIds: string[]) {
  if (!activeProjectIdRef.current) return

  const nextSessionSnapshot = reorderSessionTabs(sessionRef.current, nextOrderedTabIds)
  if (nextSessionSnapshot === sessionRef.current) return
  setSession(nextSessionSnapshot)

  const localState = normalizeLocalStateSnapshot(
    (await localStateStore.getState(activeProjectIdRef.current)) as Awaited<
      ReturnType<typeof localStateStore.getState>
    > & { activeMode: 'regular' | 'split' | 'read' | 'edit'; lastKnownScrollTop?: number },
  )

  await localStateStore.saveState(activeProjectIdRef.current, {
    ...localState,
    openDocumentPaths: nextSessionSnapshot.tabs.map((tab) => tab.documentPath),
    activeDocumentPath: getActiveTab(nextSessionSnapshot)?.documentPath ?? null,
    activeMode: nextSessionSnapshot.mode,
    regularViewState: nextSessionSnapshot.regularViewState,
  })
}
